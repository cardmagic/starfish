  def sanitize(object)
    object.extend DRbUndumped
    
    @@log = case @@options[:log]
    when String
      Logger.new(@@options[:log])
    when Class
      @@options[:log].new
    when nil, false
      Logger.new(NULL)
    else
      @@options[:log]
    end
    
    def object.logger
      @logger ||= @@log
    end
    def object._logger
      @logger ||= @@log
    end
  end
